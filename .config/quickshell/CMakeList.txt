cmake_minimum_required(VERSION 3.16)
project(NetworkPlugin)

# Buscar Qt y sus dependencias
find_package(Qt6 REQUIRED COMPONENTS Core Gui Qml)

# Definir la librería como un módulo de QML
qt_add_library(NetworkPlugin SHARED
    NetworkManager.cpp
    NetworkManager.h
)

# Configurar el plugin para que sea reconocido por QML
target_link_libraries(NetworkPlugin PRIVATE Qt6::Core Qt6::Qml)
